programa test_factorial_main;

vars n, resultado, i : entero;

inicio {
    n = 6;
    resultado = 1;
    i = 1;

    escribe("Factorial de:");
    escribe(n);

    mientras (i < n) haz {
        i = i + 1;
        resultado = resultado * i;
    }

    escribe("Resultado:");
    escribe(resultado);
}
fin
