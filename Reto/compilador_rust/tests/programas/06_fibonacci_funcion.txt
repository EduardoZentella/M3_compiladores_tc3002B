programa test_fibonacci;

vars pos, resultado, contador : entero;

entero fibonacci(n : entero) {
    vars a, b, temp, i : entero;

    a = 0;
    b = 1;
    i = 0;

    mientras (i < n) haz {
        temp = a;
        a = b;
        b = temp + b;
        i = i + 1;
    }

    regresa a;
};

inicio {
    contador = 0;
    mientras (contador < 21) haz {
        resultado = fibonacci(contador);
        escribe(resultado);
        contador = contador + 1;
    }
}
fin
